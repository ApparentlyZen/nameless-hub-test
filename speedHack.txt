-- === FONCTIONS UI V7 ===
local function AddTgl(parent, name, color, callback)
    local b = Instance.new("TextButton", parent); b.Size = UDim2.new(1, 0, 0, 30); b.BackgroundColor3 = Color3.fromRGB(30, 30, 35); b.Text = name.." [OFF]"; b.TextColor3 = Color3.fromRGB(180, 180, 180); b.Font = Enum.Font.GothamSemibold; b.TextSize = 10; Instance.new("UICorner", b).CornerRadius = UDim.new(0, 6)
    local act = false; b.MouseButton1Click:Connect(function() act = not act; b.Text = name..(act and " [ON]" or " [OFF]"); b.BackgroundColor3 = act and color or Color3.fromRGB(30, 30, 35); callback(act) end)
end

local function AddSlider(parent, name, min, max, def, callback)
    local f = Instance.new("Frame", parent); f.Size = UDim2.new(1, 0, 0, 38); f.BackgroundTransparency = 1
    local l = Instance.new("TextLabel", f); l.Size = UDim2.new(1,0,0,15); l.Text = name.." : "..def; l.TextColor3 = Color3.new(0.7,0.7,0.7); l.Font = Enum.Font.Gotham; l.TextSize = 10; l.BackgroundTransparency = 1
    local b = Instance.new("Frame", f); b.Size = UDim2.new(1,0,0,4); b.Position = UDim2.new(0,0,0.7,0); b.BackgroundColor3 = Color3.fromRGB(40,40,45)
    local fi = Instance.new("Frame", b); fi.Size = UDim2.new((def-min)/(max-min), 0, 1, 0); fi.BackgroundColor3 = Color3.fromRGB(100, 100, 255)
    f.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then local m = uis.InputChanged:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch then local p = math.clamp((i.Position.X - b.AbsolutePosition.X) / b.AbsoluteSize.X, 0, 1); fi.Size = UDim2.new(p,0,1,0); local v = math.floor(min + (max - min) * p); l.Text = name.." : "..v; callback(v) end end) i.Changed:Connect(function() if i.UserInputState == Enum.UserInputState.End then m:Disconnect() end end) end end)
end

-- === LOGIQUE SPEED HACK ===
runService.Heartbeat:Connect(function()
    if not settings.speedActive then return end
    local char = lp.Character; local hrp = char and char:FindFirstChild("HumanoidRootPart"); local hum = char and char:FindFirstChild("Humanoid")
    if not hrp or not hum then return end

    local moveDir = hum.MoveDirection
    if settings.mode == "WalkSpeed" then
        hum.WalkSpeed = settings.speedValue
    elseif settings.mode == "CFrame" and moveDir.Magnitude > 0 then
        hrp.CFrame = hrp.CFrame + (moveDir * (settings.speedValue / 100))
    elseif settings.mode == "Velocity" and moveDir.Magnitude > 0 then
        hrp.Velocity = Vector3.new(moveDir.X * settings.speedValue, hrp.Velocity.Y, moveDir.Z * settings.speedValue)
    elseif settings.mode == "TP-Flash" and moveDir.Magnitude > 0 then
        hrp.CFrame = hrp.CFrame * CFrame.new(0, 0, -settings.speedValue/50)
    elseif settings.mode == "AnimSpeed" then
        hum.WalkSpeed = settings.speedValue
        for _, track in pairs(hum:GetPlayingAnimationTracks()) do track:AdjustSpeed(settings.speedValue / 16) end
    end
end)
