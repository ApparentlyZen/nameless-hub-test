-- === SMART PAUSE (DÉTECTION DES TOUCHES) ===
local skillKeys = {
    [Enum.KeyCode.Z] = true, [Enum.KeyCode.X] = true,
    [Enum.KeyCode.C] = true, [Enum.KeyCode.V] = true, [Enum.KeyCode.F] = true
}
UserInputService.InputBegan:Connect(function(i, p) if not p and skillKeys[i.KeyCode] then _G.IsCharging = true end end)
UserInputService.InputEnded:Connect(function(i, _) if skillKeys[i.KeyCode] then _G.IsCharging = false end end)

-- === LOGIQUE D'ATTAQUE MAX SPEED ===
local function FastAttack(target)
    if _G.IsCharging or not target then return end
    pcall(function()
        local char = LocalPlayer.Character
        local tool = char and char:FindFirstChildOfClass("Tool")
        local Net = ReplicatedStorage:WaitForChild("Modules"):WaitForChild("Net")
        if tool then
            local rEvent = tool:FindFirstChild("RemoteEvent")
            local lClick = tool:FindFirstChild("LeftClickRemote")
            if rEvent then rEvent:FireServer(false) end
            if lClick then lClick:FireServer() end
        end
        local v106 = v104 .. tostring(coroutine.running()):sub(11, 15)
        local hitPart = target:FindFirstChild("Head") or target:FindFirstChild("HumanoidRootPart")
        if hitPart then
            Net["RE/RegisterAttack"]:FireServer(0)
            Net["RE/RegisterHit"]:FireServer(hitPart, {}, v106)
        end
    end)
end

local function AddTgl(parent, name, color, callback)
    local b = Instance.new("TextButton", parent); b.Size = UDim2.new(1, 0, 0, 30); b.BackgroundColor3 = Color3.fromRGB(30, 30, 35); b.Text = name.." [OFF]"; b.TextColor3 = Color3.fromRGB(180, 180, 180); b.Font = Enum.Font.GothamSemibold; b.TextSize = 10
    Instance.new("UICorner", b).CornerRadius = UDim.new(0, 6)
    local act = false; b.MouseButton1Click:Connect(function() act = not act; b.Text = name..(act and " [ON]" or " [OFF]"); b.BackgroundColor3 = act and color or Color3.fromRGB(30, 30, 35); b.TextColor3 = act and Color3.new(1,1,1) or Color3.fromRGB(180,180,180); callback(act) end)
end

local function AddSlider(parent, name, min, max, def, callback)
    local f = Instance.new("Frame", parent); f.Size = UDim2.new(1, 0, 0, 38); f.BackgroundTransparency = 1
    local l = Instance.new("TextLabel", f); l.Size = UDim2.new(1,0,0,15); l.Text = name.." : "..def; l.TextColor3 = Color3.new(0.7,0.7,0.7); l.Font = Enum.Font.Gotham; l.TextSize = 10; l.BackgroundTransparency = 1
    local b = Instance.new("Frame", f); b.Size = UDim2.new(1,0,0,4); b.Position = UDim2.new(0,0,0.7,0); b.BackgroundColor3 = Color3.fromRGB(40,40,45)
    local fi = Instance.new("Frame", b); fi.Size = UDim2.new((def-min)/(max-min), 0, 1, 0); fi.BackgroundColor3 = Color3.fromRGB(100, 100, 255)
    f.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then local m = UserInputService.InputChanged:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch then local p = math.clamp((i.Position.X - b.AbsolutePosition.X) / b.AbsoluteSize.X, 0, 1); fi.Size = UDim2.new(p,0,1,0); local v = math.floor(min + (max - min) * p); l.Text = name.." : "..v; callback(v) end end) i.Changed:Connect(function() if i.UserInputState == Enum.UserInputState.End then m:Disconnect() end end) end end)
end

local function UpdatePlayerList()
    for _,v in pairs(DScroll:GetChildren()) do if v:IsA("TextButton") then v:Destroy() end end
    local function createBtn(n, d)
        local b = Instance.new("TextButton", DScroll); b.Size = UDim2.new(1,0,0,22); b.Text = d; b.BackgroundColor3 = Color3.fromRGB(30,30,40); b.TextColor3 = Color3.new(1,1,1); b.TextSize = 9
        b.MouseButton1Click:Connect(function() settings.selectedPlayer = n; DBtn.Text = "Cible: "..d; DScroll.Visible = false end)
    end
    createBtn("Closest", "Closest")
    for _,p in pairs(Players:GetPlayers()) do if p ~= LocalPlayer then createBtn(p.Name, p.DisplayName) end end
end
UpdatePlayerList()
RefreshBtn.MouseButton1Click:Connect(UpdatePlayerList)
